function renderPage(url) { // Use a CORS proxy to load external URLs const corsProxy = 'https://cors-anywhere.herokuapp.com/'; const proxiedUrl = corsProxy + url; // Create an iframe to display the content const iframe = document.createElement('iframe'); iframe.src = proxiedUrl; iframe.width = '100%'; iframe.height = '600px'; iframe.style.border = 'none'; // Clear the current page and append the iframe const content = document.getElementById('content'); content.innerHTML = ''; content.appendChild(iframe); }
